! =============================================================
! SW-ACC-1 — Access Switch 1
! Role: VLAN 10 (Staff) access ports, trunk uplink to SW-DIST
! Port Security, BPDU Guard, Rapid PVST+
! =============================================================
!
version 15.9
hostname SW-ACC-1
!
no service finger
no service udp-small-servers
service password-encryption
!
ip domain-name lab.internal
crypto key generate rsa modulus 2048
ip ssh version 2
ip ssh time-out 60
ip ssh authentication-retries 3
!
username admin privilege 15 algorithm-type scrypt secret <CHANGE-ME>
!
banner motd #
*************************************************************
*  AUTHORIZED ACCESS ONLY - ALL ACTIVITY IS MONITORED      *
*************************************************************
#
!
! ---- VLAN Database ----
vlan 10
 name STAFF
vlan 20
 name SERVERS
vlan 30
 name GUEST
vlan 99
 name MGMT
!
! ---- Staff Access Ports (Gi0/1 - Gi0/8) ----
! Port Security: max 2 MACs, sticky learning, restrict violation
! BPDU Guard: shut down port if a BPDU is received (prevents switch spoofing)
! PortFast: skip STP listening/learning for end devices
interface range GigabitEthernet0/1 - 8
 description STAFF-WORKSTATION
 switchport mode access
 switchport access vlan 10
 !
 ! Port Security (Security+ - Hardening, CCNA - Security Fundamentals)
 switchport port-security
 switchport port-security maximum 2
 switchport port-security violation restrict
 switchport port-security mac-address sticky
 !
 spanning-tree portfast
 spanning-tree bpduguard enable
 !
 no shutdown
!
! ---- Unused Ports: Shut down and assign to MGMT VLAN ----
! Security best practice: unused ports are disabled
interface range GigabitEthernet0/9 - 23
 description UNUSED-DISABLED
 switchport mode access
 switchport access vlan 99
 shutdown
!
! ---- Uplink Trunk to SW-DIST ----
! Root Guard: prevents this port from becoming the STP root
interface GigabitEthernet0/24
 description UPLINK-TO-SW-DIST-Gi0/2
 switchport trunk encapsulation dot1q
 switchport mode trunk
 switchport trunk allowed vlan 10,20,30,99
 switchport trunk native vlan 99
 spanning-tree guard root
 no shutdown
!
! ---- Management SVI (VLAN 99) ----
interface Vlan99
 description OUT-OF-BAND-MANAGEMENT
 ip address 10.10.99.11 255.255.255.0
 no shutdown
!
! Default gateway for management traffic
ip default-gateway 10.10.99.1
!
! ---- Spanning Tree ----
! Rapid PVST+ for fast convergence (sub-2-second vs 30-50 sec for 802.1D)
spanning-tree mode rapid-pvst
spanning-tree loopguard default
!
! ---- VTY Lines (SSH only) ----
line vty 0 4
 login local
 transport input ssh
 exec-timeout 5 0
 logging synchronous
!
line console 0
 login local
 exec-timeout 10 0
!
no ip http server
no ip http secure-server
no cdp run
!
! ---- SNMPv3 ----
snmp-server group MONITORING v3 priv
snmp-server user wazuh-monitor MONITORING v3 auth sha Auth@2026 priv aes 128 Priv@2026
snmp-server host 10.10.20.10 version 3 priv wazuh-monitor
!
! ---- NTP ----
ntp server 10.10.20.10
!
! ---- Logging to Wazuh SIEM ----
logging host 10.10.20.10 transport udp port 514
logging trap informational
logging source-interface Vlan99
logging buffered 8192 informational
!
end
!
! ---- Verification Commands ----
! show vlan brief
! show interfaces trunk
! show port-security interface gi0/1
! show port-security address
! show spanning-tree vlan 10
! show spanning-tree interface gi0/24 detail
