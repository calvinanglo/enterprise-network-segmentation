! R1-CORE - Core Router
! WAN edge, static default to pfSense, OSPF area 0
!
version 15.9
hostname R1-CORE
!
no service finger
no service udp-small-servers
no service tcp-small-servers
service password-encryption
!
ip domain-name lab.internal
!
! need domain-name set before you can generate the RSA key
crypto key generate rsa modulus 2048
ip ssh version 2
ip ssh time-out 60
ip ssh authentication-retries 3
ip ssh logging events
!
username admin privilege 15 algorithm-type scrypt secret <CHANGE-ME>
!
banner motd #
*************************************************************
* AUTHORIZED ACCESS ONLY - ALL ACTIVITY IS MONITORED       *
*************************************************************
#
!
interface GigabitEthernet0/0
 description WAN to pfSense
 ip address 10.0.0.2 255.255.255.0
 no shutdown
!
interface GigabitEthernet0/1
 description LAN to SW-DIST Gi0/1
 ip address 10.0.1.1 255.255.255.252
 no shutdown
!
ip route 0.0.0.0 0.0.0.0 10.0.0.1 name DEFAULT-TO-PFSENSE
!
router ospf 1
 router-id 1.1.1.1
 network 10.0.0.0 0.0.0.255 area 0
 network 10.0.1.0 0.0.0.3 area 0
 ! pushing the default route into OSPF so all VLANs get internet
 default-information originate
!
line vty 0 4
 login local
 transport input ssh
 exec-timeout 5 0
 logging synchronous
!
line console 0
 login local
 exec-timeout 10 0
 logging synchronous
!
no ip http server
no ip http secure-server
no cdp run
!
! SNMPv3 only, no community strings
snmp-server group MONITORING v3 priv
snmp-server user wazuh-monitor MONITORING v3 auth sha Auth@2026 priv aes 128 Priv@2026
snmp-server host 10.10.20.10 version 3 priv wazuh-monitor
snmp-server contact noc@lab.internal
snmp-server location Lab-Rack-A1
!
ntp server 10.10.20.10
!
logging host 10.10.20.10 transport udp port 514
logging trap informational
logging buffered 16384 informational
logging source-interface GigabitEthernet0/1
!
end
!
